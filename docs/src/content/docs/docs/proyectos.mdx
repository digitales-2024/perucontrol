## Crear proyecto

<Uml code={`
@startuml;
|Proyecto|
start
if (Crear proyecto sin cotización?) then (si)
	:Crear proyecto;
	:Obtener cliente;
else (no)
	if (Hay proyecto aprobado?) then (si)
		:Selecctionar cotización;
		:Crear proyecto;
		:Obtener datos del cliente;
	else (no)
		|Cotización|
		:Aprobar cotización;
		end
	endif
endif

|Proyecto|
:Seleccionar dirección del servicio;
:Seleccionar servicio(s) a realizar;

if (Tiene orden de servicio?) then (si)
	:Registrar Nº de orden;
else (no)
endif

if (Registrar insumos*?) then (si)
	:Registrar insumos;
else (no)
endif

:Marcar proyecto como pendiente;

end
@enduml;
`} />


## Marcar como completado

<Uml code={`
@startuml
start

if (Hay proyecto pendiente?) then (si)
	:Marcar como completado;
	end
else (no)
	:Crear proyecto;
	end
endif
@enduml
`} />
